        // ---- PARSE DEPENDENCIES ----

        function parseDependencies(files) {

            const deps = [];


            if (files['package.json']) {

                try {

                    const pkg = JSON.parse(files['package.json']);

                    const allDeps = { ...pkg.dependencies, ...pkg.devDependencies };

                    for (const [name, ver] of Object.entries(allDeps).slice(0, 20)) {

                        deps.push({ name, version: ver, source: 'npm' });

                    }

                } catch (e) {}

            }


            const gradleFile = Object.keys(files).find(f => f.includes('build.gradle'));

            if (gradleFile && files[gradleFile]) {

                const content = files[gradleFile];

                const implMatches = content.match(/implementation\s+['"]([^'"]+)['"]/g) || [];

                for (const match of implMatches.slice(0, 15)) {

                    const dep = match.replace(/implementation\s+['"]/, '').replace(/['"]/, '');

                    deps.push({ name: dep, version: '', source: 'gradle' });

                }

            }


            if (files['requirements.txt']) {

                const lines = files['requirements.txt'].split('\n').filter(l => l.trim() && !l.startsWith('#'));

                for (const line of lines.slice(0, 15)) {

                    const parts = line.split('==');

                    deps.push({ name: parts[0], version: parts[1] || '', source: 'pip' });

                }

            }


            return deps;

        }

        // ---- FETCH REPO ----

        async function fetchRepo(owner, repo) {

            logMsg(`📡 Fetching ${owner}/${repo}...`, 'info');


            const repoRes = await fetch(`https://api.github.com/repos/${owner}/${repo}`);

            if (!repoRes.ok) {

                throw new Error(`HTTP ${repoRes.status}: ${repoRes.statusText}`);

            }

            const repoData = await repoRes.json();

            const branch = repoData.default_branch || 'main';

            logMsg(`📌 Default branch: ${branch}`, 'info');


            const treeRes = await fetch(

                `https://api.github.com/repos/${owner}/${repo}/git/trees/${branch}?recursive=1`

            );

            if (!treeRes.ok) {

                throw new Error(`Tree fetch failed: ${treeRes.status}`);

            }

            const treeData = await treeRes.json();


            const filePaths = treeData.tree

                .filter(item => item.type === 'blob')

                .map(item => item.path)

                .filter(path => {

                    const skip = ['.png', '.jpg', '.jpeg', '.gif', '.ico', '.svg', '.pdf', '.zip', '.tar', '.gz', '.mp4',

                        '.mp3', '.wav', '.webp', '.apk', '.aab'

                    ];

                    const ext = path.substring(path.lastIndexOf('.')).toLowerCase();

                    if (skip.includes(ext)) return false;

                    if (path.includes('node_modules/') || path.includes('/dist/') ||

                        path.includes('/build/') || path.includes('.git/')) return false;

                    return true;

                });


            logMsg(`📁 Found ${filePaths.length} files (fetching first 50)`, 'info');


            const MAX_FILES = 50;

            const filesToFetch = filePaths.slice(0, MAX_FILES);

            const fileContents = {};


            for (const path of filesToFetch) {

                try {

                    const rawRes = await fetch(

                        `https://raw.githubusercontent.com/${owner}/${repo}/${branch}/${path}`

                    );

                    if (rawRes.ok) {

                        fileContents[path] = await rawRes.text();

                    }

                } catch (e) {

                    // Skip files that fail

                }

                await new Promise(r => setTimeout(r, 30));

            }


            logMsg(`✅ Loaded ${Object.keys(fileContents).length} files`, 'success');

            return fileContents;

        }


        // ---- RENDER FILE TREE ----

        function renderTree(files) {

            const paths = Object.keys(files);

            if (paths.length === 0) {

                explorer.innerHTML =

                    `<div style="color:var(--text-muted); padding:20px; text-align:center;">No files found</div>`;

                return;

            }


            const root = { children: {} };

            for (const path of paths.sort()) {

                const parts = path.split('/');

                let current = root;

                for (let i = 0; i < parts.length; i++) {

                    const part = parts[i];

                    if (!current.children[part]) {

                        current.children[part] = {

                            name: part,

                            isFile: i === parts.length - 1,

                            children: {},

                            fullPath: path

                        };

                    }

                    current = current.children[part];

                }

            }


            function renderNode(node, indent = 0) {

                const entries = Object.values(node.children).sort((a, b) => {

                    if (a.isFile !== b.isFile) return a.isFile ? 1 : -1;

                    return a.name.localeCompare(b.name);

                });


                let html = '';

                for (const child of entries) {

                    if (child.isFile) {

                        const icon = getFileIcon(child.name);

                        html += `

                            <li style="padding-left:${indent + 20}px;">

                                <span class="file-item" data-path="${child.fullPath}">

                                    ${icon} ${child.name}

                                </span>

                            </li>

                        `;

                    } else {

                        html += `

                            <li style="padding-left:${indent + 8}px;">

                                <span class="folder-item">📁 ${child.name}</span>

                                <ul class="file-tree">${renderNode(child, indent + 20)}</ul>

                            </li>

                        `;

                    }

                }

                return html;

            }


            explorer.innerHTML = `<ul class="file-tree">${renderNode(root)}</ul>`;

            console.log(`✅ Rendered ${paths.length} files in explorer`);

        }


        function getFileIcon(name) {

            const ext = name.split('.').pop().toLowerCase();

            const icons = {

                html: '🌐', htm: '🌐', css: '🎨', js: '🟨', ts: '🔵',

                jsx: '⚛️', tsx: '⚛️', vue: '🟩', json: '📋', md: '📝',

                py: '🐍', rb: '💎', go: '🐹', rs: '🦀', java: '☕',

                kt: '🔷', kts: '🔷', c: '⚙️', cpp: '⚙️', sh: '💻',

                yml: '📄', yaml: '📄', toml: '📄', lock: '🔒',

        function getFileTypeName(path) {

            const ext = path.split('.').pop().toLowerCase();

            const types = {

                html: 'HTML', htm: 'HTML', css: 'CSS', js: 'JavaScript',

                ts: 'TypeScript', jsx: 'React JSX', tsx: 'React TSX',

                vue: 'Vue.js', json: 'JSON', md: 'Markdown', py: 'Python',

                rb: 'Ruby', go: 'Go', rs: 'Rust', java: 'Java',

                kt: 'Kotlin', kts: 'Kotlin Script', c: 'C', cpp: 'C++',

                sh: 'Shell', yml: 'YAML', yaml: 'YAML', toml: 'TOML',

                lock: 'Lock File', gitignore: 'Git Ignore', txt: 'Text',

                xml: 'XML', svg: 'SVG', gradle: 'Gradle', properties: 'Properties'

            };

            return types[ext] || 'Code File';

        }


        // ---- RENDER README ----

        function renderReadme(content) {

            let html = content;

            html = html.replace(/^# (.*$)/gim, '<h1>$1</h1>');

            html = html.replace(/^## (.*$)/gim, '<h2>$1</h2>');

            html = html.replace(/^### (.*$)/gim, '<h3>$1</h3>');

            html = html.replace(/```([\s\S]*?)```/g, '<pre><code>$1</code></pre>');

            html = html.replace(/`([^`]+)`/g, '<code>$1</code>');

            html = html.replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2" target="_blank">$1</a>');

            html = html.replace(/^- (.*$)/gim, '<li>$1</li>');

            html = html.replace(/^(\d+)\. (.*$)/gim, '<li>$2</li>');

            html = html.replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>');

            html = html.replace(/\*([^*]+)\*/g, '<em>$1</em>');

            html = html.replace(/^> (.*$)/gim, '<blockquote>$1</blockquote>');

            html = html.replace(/\n/g, '<br />');

            return html;

        }

        function showFile(path, content) {

            console.log(`📂 showFile() called with: ${path} (${content ? content.length : 0} chars)`);


            if (!content) {

                console.error(`❌ No content for: ${path}`);

                logMsg(`❌ No content for: ${path}`, 'error');

                return;

            }


            currentFilePath = path;

            logMsg(`📖 Opening: ${path} (${content.length} chars)`, 'info');


            placeholder.classList.add('hidden');


            const isReadme = path.toLowerCase().includes('readme');

            const isHtml = path.match(/\.(html|htm)$/i);


            const icon = getFileIcon(path);

            const typeName = getFileTypeName(path);

            const lines = content.split('\n').length;

            const size = content.length;

            const sizeDisplay = size < 1024 ? `${size} B` : size < 1048576 ? `${(size / 1024).toFixed(1)} KB` :

                `${(size / 1048576).toFixed(1)} MB`;

            const preview = content.length > 500 ? content.substring(0, 500) + '\n... (truncated)' : content;


            fileIcon.textContent = icon;

            fileName.textContent = path.split('/').pop();

            fileType.textContent = `📂 ${typeName}`;

            fileSize.textContent = `📦 ${sizeDisplay}`;

            fileLines.textContent = `📝 ${lines} lines`;

            filePreviewContent.textContent = preview;


            if (isReadme) {

                readmeView.innerHTML = renderReadme(content);

                switchTab('readme');

                logMsg(`📖 Displaying README`, 'success');

                return;

            }


            if (isHtml) {

                try {

                    const blob = new Blob([content], { type: 'text/html' });

                    const url = URL.createObjectURL(blob);

                    previewFrame.src = url;

                    switchTab('preview');

                    logMsg(`✅ Rendered ${path} in preview`, 'success');

                    setTimeout(() => URL.revokeObjectURL(url), 1000);

                } catch (e) {

                    logMsg(`❌ Failed to render: ${e.message}`, 'error');

                }

                return;

            }


            codeView.textContent = `// ${path}\n\n${content}`;

            switchTab('code');

            filePreviewCard.classList.add('active');

            logMsg(`📄 Showing code for ${path} (also showing preview card)`, 'info');

        }
